## 9.2 Docker Compose 配置

```yaml
# docker-compose.yml
version: '3.8'

services:
  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/conf.d/default.conf
      - ./nginx/ssl:/etc/nginx/ssl
      - static_volume:/static
      - media_volume:/media
    depends_on:
      - django
      - frontend

  postgres:
    image: postgres:15-alpine
    volumes:
      - postgres_data:/var/lib/postgresql/data
    environment:
      POSTGRES_DB: ${DB_NAME:-styleflow}
      POSTGRES_USER: ${DB_USER:-styleflow}
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${DB_USER:-styleflow}"]
      interval: 10s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    volumes:
      - redis_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s

  django:
    build:
      context: ./backend
      dockerfile: Dockerfile
    volumes:
      - static_volume:/app/static
      - media_volume:/app/media
    env_file:
      - .env
    environment:
      DJANGO_SETTINGS_MODULE: config.settings.prod
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
    command: >
      sh -c "python manage.py migrate &&
             python manage.py collectstatic --noinput &&
             gunicorn config.wsgi:application --bind 0.0.0.0:8000 --workers 4"

  celery-worker:
    build: ./backend
    env_file:
      - .env
    environment:
      DJANGO_SETTINGS_MODULE: config.settings.prod
    depends_on:
      - django
      - redis
    command: celery -A celery_app worker -l info -Q ai_generation,llm_tasks,sync,default

  celery-beat:
    build: ./backend
    env_file:
      - .env
    environment:
      DJANGO_SETTINGS_MODULE: config.settings.prod
    depends_on:
      - django
      - redis
    command: celery -A celery_app beat -l info

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    env_file:
      - .env
    environment:
      NEXT_PUBLIC_API_URL: https://api.styleflow.com/api/v1

  minio:
    image: minio/minio:latest
    volumes:
      - minio_data:/data
    environment:
      MINIO_ROOT_USER: ${MINIO_USER:-minioadmin}
      MINIO_ROOT_PASSWORD: ${MINIO_PASSWORD}
    command: server /data --console-address ":9001"

volumes:
  postgres_data:
  redis_data:
  static_volume:
  media_volume:
  minio_data:
```

## 9.3 CI/CD 流水线

### GitHub Actions

```yaml
# .github/workflows/deploy.yml
name: Deploy

on:
  push:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Backend tests
        run: |
          cd backend
          pip install -r requirements/dev.txt
          pytest --cov=apps --cov-report=term-missing

      - name: Frontend lint
        run: |
          cd frontend
          npm ci
          npm run lint

  deploy:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Deploy to server
        env:
          SSH_PRIVATE_KEY: ${{ secrets.SSH_PRIVATE_KEY }}
          SERVER_HOST: ${{ secrets.SERVER_HOST }}
        run: |
          ssh -o StrictHostKeyChecking=no $SERVER_HOST '
            cd /opt/styleflow &&
            git pull &&
            docker compose pull &&
            docker compose up -d --build --remove-orphans &&
            docker system prune -f
          '
```

## 9.4 Nginx 配置

```nginx
# nginx/nginx.conf
server {
    listen 443 ssl http2;
    server_name styleflow.com api.styleflow.com;

    ssl_certificate /etc/nginx/ssl/cert.pem;
    ssl_certificate_key /etc/nginx/ssl/key.pem;

    # Django API
    location /api/ {
        proxy_pass http://django:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # SSE 支持
        proxy_buffering off;
        proxy_cache off;
        proxy_read_timeout 86400s;
    }

    # Django Admin / OpenAPI
    location /admin/ { proxy_pass http://django:8000; }
    location /docs  { proxy_pass http://django:8000; }

    # Django 静态文件
    location /static/ { alias /static/; }

    # MinIO 图片代理
    location /media/ {
        proxy_pass http://minio:9000/styleflow/;
        proxy_set_header Host $host;
    }

    # Next.js 前端
    location / {
        proxy_pass http://frontend:3000;
        proxy_set_header Host $host;
    }
}

server {
    listen 80;
    server_name styleflow.com api.styleflow.com;
    return 301 https://$host$request_uri;
}
```

## 9.5 监控与日志

### 日志方案（轻量级）

```
Django/Celery → 文件日志 → logrotate 轮转
                          → 错误日志发送 Sentry
```

```python
# config/settings/prod.py
LOGGING = {
    'version': 1,
    'disable_existing_loggers': False,
    'formatters': {
        'json': {
            'format': '{"time":"%(asctime)s","level":"%(levelname)s","module":"%(module)s","message":"%(message)s"}',
        },
    },
    'handlers': {
        'console': {
            'class': 'logging.StreamHandler',
            'formatter': 'json',
        },
        'file': {
            'class': 'logging.handlers.RotatingFileHandler',
            'filename': '/var/log/styleflow/django.log',
            'maxBytes': 10485760,  # 10MB
            'backupCount': 10,
            'formatter': 'json',
        },
    },
    'root': {
        'handlers': ['console', 'file'],
        'level': 'INFO',
    },
}
```

### 监控指标（Prometheus + Grafana 可选）

| 指标 | 说明 |
|------|------|
| `django_requests_total` | 总请求数（按状态码、路径） |
| `django_request_duration` | 请求耗时分布 |
| `celery_tasks_total` | 任务总数（按状态、队列） |
| `celery_task_duration` | 任务耗时分布 |
| `db_connection_pool` | 数据库连接池状态 |
| `ai_call_total` | AI 调用次数（按模型、成功/失败） |
| `ai_call_cost` | AI 调用费用累计 |

## 9.6 备份策略

| 数据 | 备份方式 | 频率 | 保留 |
|------|---------|------|------|
| PostgreSQL | `pg_dump` → 压缩 → OSS | 每日 04:00 | 30 天 |
| MinIO 图片 | `mc mirror` → OSS | 实时同步 | 90 天 |
| Redis (RDB) | `redis-cli save` → OSS | 每 6 小时 | 7 天 |
| Django 配置 | Git（.env 除外） | 每次提交 | 永久 |

### 备份脚本

```bash
#!/bin/bash
# scripts/backup.sh

TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/backups"

# 数据库备份
PGPASSWORD=$DB_PASSWORD pg_dump -h $DB_HOST -U $DB_USER $DB_NAME \
    | gzip > $BACKUP_DIR/db_$TIMESTAMP.sql.gz

# 上传到 OSS
ossutil cp $BACKUP_DIR/db_$TIMESTAMP.sql.gz oss://styleflow-backups/db/

# 清理本地旧文件（保留 3 天）
find $BACKUP_DIR -name "*.sql.gz" -mtime +3 -delete
```

## 9.8 服务器推荐配置

| 配置 | 最低 | 推荐 | 说明 |
|------|------|------|------|
| **CPU** | 4 核 | 8 核 | 多核用于 Gunicorn workers + Celery |
| **内存** | 8 GB | 16 GB | Django + Redis + Next.js 均需内存 |
| **硬盘** | 100 GB SSD | 500 GB SSD | 图片存储占空间 |
| **GPU** | 无（使用云端 API） | RTX 3090 / A10 (24GB VRAM) | 本地运行 SD + VTON |
| **带宽** | 10 Mbps | 100 Mbps | 图片上传下载 |
