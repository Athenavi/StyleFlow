# 05 - AI 服务集成设计

> **版本**：v1.0 | **更新**：2026-06-26

---

## 5.1 设计目标

- **统一抽象**：所有 AI 模型通过统一的 `AIService` 接口调用，切换模型只需改配置
- **异步优先**：AI 调用耗时 > 1s 的全部走 Celery 异步队列
- **可观测**：记录每次调用的 Token/费用、耗时、成功率
- **优雅降级**：主模型失败自动切换到备用模型

## 5.3 LLM 服务封装

### 接口定义

```python
# common/aiservice/llm/base.py
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Optional

@dataclass
class LLMResponse:
    content: str
    model: str
    tokens_used: int
    cost: float
    latency_ms: int

class BaseLLMService(ABC):
    @abstractmethod
    def chat(self, messages: list[dict], **kwargs) -> LLMResponse:
        """对话补全"""
        ...

    @abstractmethod
    def extract_json(self, prompt: str, schema: dict) -> dict:
        """结构化输出（Function Calling）"""
        ...

    def generate_text(self, prompt: str, max_tokens: int = 1024) -> str:
        """简化的文本生成"""
        return self.chat([{"role": "user", "content": prompt}],
                        max_tokens=max_tokens).content
```

### OpenAI 实现

```python
# common/aiservice/llm/openai.py
import openai
from .base import BaseLLMService, LLMResponse

class OpenAIService(BaseLLMService):
    def __init__(self, api_key: str, model: str = "gpt-4o"):
        self.client = openai.OpenAI(api_key=api_key)
        self.model = model

    def chat(self, messages: list[dict], **kwargs) -> LLMResponse:
        import time
        start = time.time()
        resp = self.client.chat.completions.create(
            model=self.model, messages=messages, **kwargs
        )
        usage = resp.usage
        cost = self._calculate_cost(usage)
        return LLMResponse(
            content=resp.choices[0].message.content,
            model=self.model,
            tokens_used=usage.total_tokens,
            cost=cost,
            latency_ms=int((time.time() - start) * 1000),
        )

    def _calculate_cost(self, usage):
        # 按模型定价表计算
        pricing = {"gpt-4o": (2.5, 10), "gpt-4o-mini": (0.15, 0.6)}  # input, output per 1M tokens
        input_cost = usage.prompt_tokens / 1_000_000 * pricing[self.model][0]
        output_cost = usage.completion_tokens / 1_000_000 * pricing[self.model][1]
        return round(input_cost + output_cost, 6)
```

```python
        ...

    @abstractmethod
    def img2img(self, init_image_url: str, prompt: str, **kwargs) -> ImageResult:
        ...
```

### Stable Diffusion (A1111 WebUI API)

```python
class SDWebUIService(BaseImageService):
    def __init__(self, base_url: str):
        self.base_url = base_url.rstrip('/')

    def text2img(self, prompt, negative_prompt="", width=512, height=768, **kwargs):
        import time, requests
        start = time.time()
        payload = {
            "prompt": prompt,
            "negative_prompt": negative_prompt or self._default_neg(),
            "width": width,
            "height": height,
            "steps": kwargs.get("steps", 25),
            "cfg_scale": kwargs.get("cfg_scale", 7),
            "seed": kwargs.get("seed", -1),
        }
        resp = requests.post(f"{self.base_url}/sdapi/v1/txt2img", json=payload, timeout=120)
        data = resp.json()
        # 保存图片到 MinIO，返回 URL
        image_url = self._save_to_storage(data['images'][0])
        return ImageResult(
            image_url=image_url,
            seed=data.get('seed', -1),
            model='sd-xl',
            latency_ms=int((time.time() - start) * 1000),
        )

    def _default_neg(self):
        return "lowres, bad anatomy, bad hands, text, error, extra digit, fewer digits, cropped, worst quality, low quality, normal quality, jpeg artifacts, signature, watermark, username, blurry"
```

## 5.6 Celery 任务编排

```python
# apps/design/tasks.py
from celery import shared_task
from common.aiservice import get_image_service

@shared_task(bind=True, max_retries=3, default_retry_delay=30)
def generate_design_task(self, user_id: int, prompt: str, category: str,
                          width: int, height: int):
    """AI 设计稿生成任务"""
    try:
        service = get_image_service('default')
        result = service.text2img(prompt=prompt, width=width, height=height)

        # 保存到数据库
        design = Design.objects.create(
            creator_id=user_id,
            prompt=prompt,
            category=category,
            image_url=result.image_url,
            width=width,
            height=height,
            metadata={'seed': result.seed, 'model': result.model},
        )
        # 发送 SSE 通知
        notify_task_completed(user_id, design.id, result.image_url)
        return {'design_id': design.id, 'image_url': result.image_url}

    except requests.Timeout:
        # 超时重试，切换到备用模型
        if self.request.retries < self.max_retries:
            raise self.retry()
        else:
            # 最终失败，通知用户
            notify_task_failed(user_id, "生成超时，请稍后重试")
            return {'error': 'timeout'}
```

### 任务类型与队列配置

| 任务 | 队列 | 优先级 | 超时 | 说明 |
|------|------|--------|------|------|
| `generate_design_task` | `ai_generation` | 中 | 300s | SD 文生图 |
| `img2img_task` | `ai_generation` | 中 | 300s | SD 图生图 |

## 5.7 成本追踪与限流

```python
# common/aiservice/cost_tracker.py
class AICostTracker:
    """AI 调用成本追踪"""

    def record_call(self, user_id, service_type: str, model: str,
                     tokens: int, cost: float, latency: int, success: bool):
        """记录一次 AI 调用"""
        AICallLog.objects.create(
            user_id=user_id,
            service_type=service_type,
            model=model,
            tokens_used=tokens,
            cost=cost,
            latency_ms=latency,
            success=success,
        )

    def get_user_usage(self, user_id, days=30):
        """用户用量统计"""
        return AICallLog.objects.filter(
            user_id=user_id,
            created_at__gte=timezone.now() - timedelta(days=days),
        ).aggregate(
            total_calls=Count('id'),
            total_cost=Sum('cost'),
            total_tokens=Sum('tokens_used'),
        )
```

```python
# common/aiservice/image/vton.py
class VTONService:
    def __init__(self, base_url: str):
        self.base_url = base_url

    def synthesize(self, person_image_url: str, garment_image_url: str,
                   category: str = "upper") -> ImageResult:
        """提交试衣合成"""
        import time, requests
        start = time.time()
        resp = requests.post(
            f"{self.base_url}/generate",
            json={
                "person_image": person_image_url,
                "garment_image": garment_image_url,
                "category": category,
            },
            timeout=120,
        )
        result = resp.json()
        return ImageResult(
            image_url=self._save_to_storage(result['output']),
            seed=0,
            model='idm-vton',
            latency_ms=int((time.time() - start) * 1000),
        )
```
